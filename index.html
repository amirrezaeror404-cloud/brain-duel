const express = require("express"); const http = require("http"); const { Server } = require("socket.io"); const cors = require("cors"); const app = express(); app.use(cors()); app.use(express.static("public")); const server = http.createServer(app); const io = new Server(server); // ---------------- DATA ---------------- let waiting = []; let matches = {}; // ---------------- QUESTIONS ---------------- const QUESTIONS = [ { q: "Capital of France?", options: ["Berlin","Paris","Rome","London"], a: 1 }, { q: "2 + 2 = ?", options: ["3","4","5","6"], a: 1 }, { q: "Color of the sky?", options: ["Green","Blue","Red","Yellow"], a: 1 }, { q: "5 * 2 = ?", options: ["10","12","8","15"], a: 0 }, { q: "Capital of Iran?", options: ["Tehran","Mashhad","Shiraz","Tabriz"], a: 0 }, { q: "10 / 2 = ?", options: ["2","3","5","6"], a: 2 }, ]; // ---------------- HELPERS ---------------- function shuffle(arr) { return [...arr].sort(() => Math.random() - 0.5); } function getQuestions() { return shuffle(QUESTIONS).slice(0, 5); } // ---------------- SOCKET ---------------- io.on("connection", (socket) => { console.log("User connected:", socket.id); // FIND MATCH socket.on("find_match", (user) => { waiting.push({ socket, user }); if (waiting.length >= 2) { const p1 = waiting.shift(); const p2 = waiting.shift(); const matchId = Date.now().toString(); matches[matchId] = { players: [p1.user.id, p2.user.id], index: {}, scores: {}, questions: {} }; matches[matchId].players.forEach(u => { matches[matchId].index[u] = 0; matches[matchId].scores[u] = 0; matches[matchId].questions[u] = getQuestions(); }); p1.socket.join(matchId); p2.socket.join(matchId); io.to(matchId).emit("match_found", { matchId }); sendQuestion(matchId); } }); // ANSWER socket.on("answer", ({ matchId, userId, answer }) => { handleAnswer(matchId, userId, answer); }); }); // ---------------- SEND QUESTION ---------------- function sendQuestion(matchId) { const m = matches[matchId]; if (!m) return; m.players.forEach(userId => { const i = m.index[userId]; const q = m.questions[userId][i]; io.to(matchId).emit("question", { userId, q }); }); } // ---------------- HANDLE ANSWER ---------------- function handleAnswer(matchId, userId, answer) { const m = matches[matchId]; if (!m) return; const i = m.index[userId]; const q = m.questions[userId][i]; if (answer === q.a) { m.scores[userId]++; } m.index[userId]++; if (m.index[userId] >= m.questions[userId].length) { endGame(matchId); return; } sendQuestion(matchId); } // ---------------- END GAME ---------------- function endGame(matchId) { const m = matches[matchId]; const [p1, p2] = m.players; let winner = "draw"; if (m.scores[p1] > m.scores[p2]) winner = p1; if (m.scores[p2] > m.scores[p1]) winner = p2; io.to(matchId).emit("match_end", { winner, scores: m.scores }); delete matches[matchId]; } // ---------------- START SERVER ---------------- server.listen(process.env.PORT || 3000, () => { console.log("Server running..."); });
